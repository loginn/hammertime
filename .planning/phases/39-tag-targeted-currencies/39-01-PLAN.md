```gdscript
class_name TagHammer extends Currency

var tag_name: String  # Uppercase constant e.g. "FIRE", "DEFENSE"


func _init(p_tag_name: String) -> void:
    tag_name = p_tag_name
    currency_name = p_tag_name.capitalize() + " Hammer"


func can_apply(item: Item) -> bool:
    if item.rarity != Item.Rarity.NORMAL:
        return false
    return _has_matching_affixes(item)


func get_error_message(item: Item) -> String:
    if item.rarity != Item.Rarity.NORMAL:
        return currency_name + " can only be used on Normal items"
    if not _has_matching_affixes(item):
        return "No " + tag_name.to_lower() + "-tagged mods available for this item"
    return ""


func _do_apply(item: Item) -> void:
    item.rarity = Item.Rarity.RARE

    var mod_count: int = randi_range(4, 6)
    for i in range(mod_count):
        var choose_prefix: bool = randi_range(0, 1) == 0
        if choose_prefix:
            if not item.add_prefix():
                if not item.add_suffix():
                    break
        else:
            if not item.add_suffix():
                if not item.add_prefix():
                    break

    _guarantee_tag_affix(item)
    item.update_value()


func _has_matching_affixes(item: Item) -> bool:
    for prefix in ItemAffixes.prefixes:
        if item.has_valid_tag(prefix) and tag_name in prefix.tags:
            return true
    for suffix in ItemAffixes.suffixes:
        if item.has_valid_tag(suffix) and tag_name in suffix.tags:
            return true
    return false


func _guarantee_tag_affix(item: Item) -> void:
    # Check if any existing affix already satisfies the tag requirement
    for affix in item.prefixes + item.suffixes:
        if tag_name in affix.tags:
            return  # Already satisfied

    # Build candidate pool: valid for item AND has our tag, respecting tier floor
    var floor_val: int = item._get_affix_tier_floor()

    var tag_prefixes: Array[Affix] = []
    for template in ItemAffixes.prefixes:
        if item.has_valid_tag(template) and tag_name in template.tags:
            tag_prefixes.append(ItemAffixes.from_affix(template, floor_val))

    var tag_suffixes: Array[Affix] = []
    for template in ItemAffixes.suffixes:
        if item.has_valid_tag(template) and tag_name in template.tags:
            tag_suffixes.append(ItemAffixes.from_affix(template, floor_val))

    # Replace slot[0] with a guaranteed tag affix
    # Prefer prefix if candidates exist AND item has prefixes, else try suffix
    if not tag_prefixes.is_empty() and not item.prefixes.is_empty():
        item.prefixes[0] = tag_prefixes.pick_random()
    elif not tag_suffixes.is_empty() and not item.suffixes.is_empty():
        item.suffixes[0] = tag_suffixes.pick_random()
    # If both slots are empty (shouldn't happen after 4-6 mod roll), skip silently
```

Key implementation notes:
- `tag_name` uses uppercase strings matching Tag constants ("FIRE" not "fire") — the `tag_name in affix.tags` check requires uppercase to match Tag.FIRE = "FIRE"
- `_has_matching_affixes()` cross-checks BOTH `item.has_valid_tag(prefix)` AND `tag_name in prefix.tags` — both conditions required
- `_guarantee_tag_affix()` guards `item.prefixes.is_empty()` before indexing `item.prefixes[0]` (Pitfall 5)
- `ItemAffixes.from_affix(template, floor_val)` handles tier floor clamping — do not implement custom tier range logic
- The autoload for item_affixes is `ItemAffixes` (class name `Affixes`) — use `ItemAffixes.prefixes`, `ItemAffixes.from_affix()`

```gdscript
## Attempts to spend one tag currency of the given type (e.g. "fire", "cold").
## Returns true if successful, false if count is 0 or type not found.
func spend_tag_currency(tag_type: String) -> bool:
    if tag_type not in tag_currency_counts:
        return false
    if tag_currency_counts[tag_type] <= 0:
        return false
    tag_currency_counts[tag_type] -= 1
    return true


## Adds tag currencies from a drops dictionary to tag_currency_counts.
## Keys are lowercase tag type strings ("fire", "cold", etc.).
func add_tag_currencies(drops: Dictionary) -> void:
    for tag_type in drops:
        if tag_type not in tag_currency_counts:
            tag_currency_counts[tag_type] = 0
        tag_currency_counts[tag_type] += drops[tag_type]
```
