  func get_best_item(slot_name: String) -> Item:
      var slot_array: Array = GameState.crafting_inventory[slot_name]
      if slot_array.is_empty():
          return null
      var best: Item = slot_array[0]
      for i in range(1, slot_array.size()):

      return best
