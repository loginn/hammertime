## Serializes the stash dictionary into a JSON-safe format.
## Preserves null gaps (items removed by tap-to-bench).
func _serialize_stash(stash_data: Dictionary) -> Dictionary:
    var result := {}
    for slot_type in stash_data:
        var arr: Array = []
        for item in stash_data[slot_type]:
            if item != null:
                arr.append(item.to_dict())
            else:
                arr.append(null)
        result[slot_type] = arr
    return result
```

func _group_50_save_v9_round_trip() -> void:
	# CRFT-03: Save v9 round-trips stash, bench, currencies, archetype

	# Setup: known state
	GameState.initialize_fresh_game()

	# Put items in stash
	var stash_sword := Broadsword.new(8)
	stash_sword.rarity = Item.Rarity.MAGIC
	stash_sword.add_prefix()
	GameState.add_item_to_stash(stash_sword)

	var stash_helm := IronHelm.new(8)
	GameState.add_item_to_stash(stash_helm)

	# Put item on bench
	var bench_item := Dagger.new(8)
	bench_item.rarity = Item.Rarity.MAGIC
	bench_item.add_prefix()
	GameState.crafting_bench = bench_item

	# Set currency counts including alteration/regal
	GameState.currency_counts["alteration"] = 5
	GameState.currency_counts["regal"] = 3
	GameState.currency_counts["transmute"] = 10

	# Set archetype
	GameState.hero_archetype = HeroArchetype.from_id("dex")

	# Save
	var saved := SaveManager.save_game()
	assert(saved, "50a: save_game() succeeded")

	# Wipe state
	GameState.initialize_fresh_game()
	assert(GameState.stash["weapon"].size() == 0 or GameState.stash["weapon"][0] is Broadsword, "50b: state wiped before load")

	# Load
	var loaded := SaveManager.load_game()
	assert(loaded, "50c: load_game() succeeded")

	# Verify stash
	var weapon_stash: Array = GameState.stash["weapon"]
	var found_sword := false
	for item in weapon_stash:
		if item != null and item is Broadsword and item.rarity == Item.Rarity.MAGIC:
			found_sword = true
			break
	assert(found_sword, "50d: stash weapon round-tripped (Magic Broadsword)")

	var helmet_stash: Array = GameState.stash["helmet"]
	var found_helm := false
	for item in helmet_stash:
		if item != null and item is IronHelm:
			found_helm = true
			break
	assert(found_helm, "50e: stash helmet round-tripped (IronHelm)")

	# Verify bench
	assert(GameState.crafting_bench != null, "50f: bench not null after load")
	assert(GameState.crafting_bench is Dagger, "50f: bench is Dagger")
	assert(GameState.crafting_bench.rarity == Item.Rarity.MAGIC, "50f: bench rarity is MAGIC")
	assert(GameState.crafting_bench.prefixes.size() >= 1, "50f: bench has prefix")

	# Verify currencies
	assert(GameState.currency_counts["alteration"] == 5, "50g: alteration count round-tripped")
	assert(GameState.currency_counts["regal"] == 3, "50g: regal count round-tripped")
	assert(GameState.currency_counts["transmute"] == 10, "50g: transmute count round-tripped")

	# Verify archetype
	assert(GameState.hero_archetype != null, "50h: archetype not null")
	assert(GameState.hero_archetype.id == "dex", "50h: archetype id round-tripped")

	# Verify v8 rejection: manually write a v8 save and try to load it
	var v8_data := {"version": 8, "hero_equipment": {}, "currencies": {}}
	var v8_json := JSON.stringify(v8_data)
	var file := FileAccess.open("user://hammertime_save.json", FileAccess.WRITE)
	file.store_string(v8_json)
	file = null
	var v8_loaded := SaveManager.load_game()
	assert(not v8_loaded, "50i: v8 save rejected by v9 loader")

	# Clean up: reinitialize fresh game
	GameState.initialize_fresh_game()

	print("Group 50: Save v9 round-trip — PASSED")
```
